import pandas as pd
!wget https://raw.githubusercontent.com/adrianulbona/borders/master/data/countries_discretized_3.csv
df = pd.read_csv('countries_discretized_3.csv', delimiter=',', header=-1, names=['continent', 'name', 'fips', 'iso', 'geohash'], doublequote=True)
data = []
for idx, row in df.iterrows():
    for item in row['geojson'].split(','):
        data.append({'continent': row['continent'], 'name': row['name'], 'fips': row['fips'], 'iso': row['iso'], 'geohash': item})
df = pd.DataFrame(data)
df.to_csv('countries_discretized_3_modified.csv', sep=',', index=False)
